var ApcEventUtil = new Object();


/**
 * Metodo para obter o target de um event
 */
ApcEventUtil.getTarget = function( event )
{		
	var target = event.target ? event.target : event.srcElement;
	return target;
};

/**
 * Metodo para obter o keyCode de um event
 */
ApcEventUtil.getKeyCode = function( event )
{		
	if (window.event) // Internet Explorer
		return event.keyCode;
	else if(event.which) // Nestcape
		return event.which;
	return undefined;	
};

/**
 * Dispara um evento em um componente html 
 * @param controlName
 * Id do componente onde o evento deve ser disparado
 * exemplo: (txtName , form:panelGroup1:txtCodigo)
 * @param eventName
 * nome do evento a ser disparado 
 * exemplo: (click,blur,change,...) 
 */
ApcEventUtil.fireEvent = function(elementId,eventName) {	
	var control = document.getElementById(elementId);
	// On IE
	if (control.fireEvent) {
		control.fireEvent('on' + eventName);
	}	
	// On Gecko based browsers
	else if (document.createEvent) 
	{
		var evt = document.createEvent('HTMLEvents');
		if (evt.initEvent) {
			evt.initEvent(eventName, true, true);
		}
		if (control.dispatchEvent) {
			control.dispatchEvent(evt);
		}
	}	
};
//termino do script /js/js-event-util.js (hash: cpk8m #36)

var ApcInputUtil = new Object();


/**
 * Metodo para ser aplicado no onblur e onkeypress<br />
 * faz com que o campo aceite apenas numeros inteiros<br />
 * exemplo h:inputText value="#{pessoaBean.inteiro}" onchange="return ApcInputUtil.denyNonDigit(event)" onkeypress="return ApcInputUtil.denyNonDigit(event)" /
 */
ApcInputUtil.denyNonDigit = function( event )
{		
	var target = event.target ? event.target : event.srcElement;
	if( event.type == 'blur' || event.type == "change")
	{		
		str='';
		for( i=0;i<target.value.length;i++ )
		{
			ch=target.value.charAt(i);
			if( ch>='0' && ch<='9' )
				str+=ch;
		}
		target.value=str;
		return true;
	}
    var tecla;
    
    if (window.event) // Internet Explorer
            tecla = event.keyCode;
    else if(event.which) // Nestcape
            tecla = event.which;

    if ((tecla > 47 && tecla < 58)) // numeros de 0 a 9
        return true;
    else {
        if ( tecla == 8 || !tecla ) // backspace
            return true;
        else
            return false;
    }

};

/**
 * Metodo para ser aplicado no onkeypress.<br />
 * Faz com que o enter gere um evento de perda ou transferecia de foco.<br />
 * h:inputText value="#{pessoaBean.valor}" onkeypress="return ApcInputUtil.blurOnEnter(event)"<br />
 * @param event A variavel implicita event<br />
 * @param dest  (Opcional)  - O elemento html que deve receber o foco 	
 */
ApcInputUtil.blurOnEnter = function(event,dest)
{
	var keyCode;
	var input;

	if( dest != undefined )
		if( dest.focus == undefined )
			throw 'O elemento de destino nao contem o metodo focus()'; 
		
	if(event.keyCode)
		//IE
		keyCode = event.keyCode;
	else
		//Firefox
		keyCode = event.which;
    
	if(event.srcElement)
		//IE
		input = event.srcElement;
	else
		//Firefox
		input = event.target;

	//enter
	if(keyCode == 13)
	{
		input.blur();
		if( dest != undefined )	
			dest.focus();
		return false;
	}
    return true;
};
